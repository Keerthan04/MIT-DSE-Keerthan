//Write a C++ program to sort find average of a stack
#include<iostream>
using namespace std;
#define MAX_SIZE 20
class Stack
{
    private:

        int top;
        int arr[MAX_SIZE];

    public:

        Stack()
        {
            top=-1;
        }
        void push(int ele) {
            if (isfull()) {
                cout << "Stack overflow\n";
            } else {
                top++;
                arr[top] = ele;
            }
        }
         int pop() {
            if (isempty()) {
                cout << "Stack underflow\n";
            } else {
                    int element=arr[top];
                    top--;
                    return element;

                }
            }
        int isfull() {
            if (top == MAX_SIZE-1) {
                return 1;
            } else
                return 0;
        }

        int isempty() {
            if (top == -1)
                return 1;
            else
                return 0;
        }
        int peek()
        {
            if(isempty())
                cout<<"Stack underflow";
            else
                return arr[top];
        }
        void display()
        {
            if(isempty())
                cout<<"Stack underflow";
            for(int i=top;i>=0;i--)
            {
                cout<<arr[i]<<" ";
            }
            cout<<endl;
        }
        void average() {
            Stack t;double sum=0,length=0;int ele;
            while (!isempty()) {
                ele=pop();
                sum=sum+ele;
                t.push(ele);
                length++;
        }
        double avg=sum/length;
        cout<<"the average of the array is: "<<avg<<endl;
        while(!t.isempty())
        {
            push(t.pop());
        }
        display();
        }



};
int main()
{
    Stack s;
    int n,a[100];
    cout<<"Enter number of elements to be pushed into stack"<<endl;
    cin>>n;
    cout<<"Enter elements to be pushed"<<endl;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
        s.push(a[i]);
    }
    cout<<"The stack elements are"<<endl;
    s.display();
    s.average();

}
