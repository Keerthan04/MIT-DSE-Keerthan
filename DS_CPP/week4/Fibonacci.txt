//Fibonacci series
#include<iostream>
using namespace std;
int fib(int n);
int main()
{
        int num;
        cout<<"Enter the term \n";
        cin>>num;
        cout<<"The fibonnaci terms are: "<<endl;
        for(int i=0;i<num;i++)
        {
                int fibo=fib(i);
                cout<<fibo<<" ";
        }
        cout<<endl;
        return 0;
}
int fib(int n)
{
        if(n==0)
                return 0;
        else if(n==1)
                return 1;
        else
                return(fib(n-1)+fib(n-2));
}
